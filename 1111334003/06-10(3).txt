#include "AnimalFile.h"
#include <cstring>
#include <cstdlib>
void AnimalFile::writeData()
{
    ofstream writefile("animal.dat", ios::binary);
    if (!writefile)
    {
        cout << "animal.dat檔案無法開啟！" << endl;
        exit(1);
    }
    Animal animal;
    char choice;
    do
    {
        cout << "\n請輸入動物名稱：";
        cin >> animal.name;
        cout << "請輸入年齡：";
        cin >> animal.age;
        cout << "請輸入身高：";
        cin >> animal.height;
        writefile.write(
            reinterpret_cast<char*>(&animal),
            sizeof(Animal));
        cout << "\n是否繼續輸入(Y/N)：";
        cin >> choice;
    } while (choice == 'Y' || choice == 'y');
    writefile.close();
}
void AnimalFile::calculateAverage()
{
    ifstream readfile("animal.dat", ios::binary);
    if (!readfile)
    {
        cout << "animal.dat檔案無法開啟！" << endl;
        exit(1);
    }
    Animal animal;
    int totalAge = 0;
    int totalHeight = 0;
    int count = 0;
    while (
        readfile.read(
            reinterpret_cast<char*>(&animal),
            sizeof(Animal)))
    {
        totalAge += animal.age;
        totalHeight += animal.height;
        count++;
    }
    readfile.close();
    if (count > 0)
    {
        double avgAge =
            static_cast<double>(totalAge) / count;
        double avgHeight =
            static_cast<double>(totalHeight) / count;
        cout << "\n===== Average Information =====\n";
        cout << fixed << setprecision(2);
        cout << "Average Age = "
            << avgAge << endl;
        cout << "Average Height = "
            << avgHeight << endl;
    }
}
void AnimalFile::displayData()
{
    ifstream readfile("animal.dat", ios::binary);
    if (!readfile)
    {
        cout << "animal.dat檔案無法開啟！" << endl;
        exit(1);
    }
    Animal animal;
    cout << "\n===============================" << endl;
    cout << "      Animal Information" << endl;
    cout << "===============================" << endl;
    cout << left
        << setw(15) << "Name"
        << setw(10) << "Age"
        << setw(10) << "Height"
        << endl;
    while (
        readfile.read(
            reinterpret_cast<char*>(&animal),
            sizeof(Animal)))
    {
        cout << left
            << setw(15) << animal.name
            << setw(10) << animal.age
            << setw(10) << animal.height
            << endl;
    }
    readfile.close();
}
