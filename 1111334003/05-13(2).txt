#include <iostream>
#include "aircraft.h"
using namespace std;
aircraft::aircraft()
{
    cout << "Execute aircraft constructor\n";
    cout << "Create an aircraft:\n";
    cout << "Name: ";
    cin >> name;
    cout << "ID: ";
    cin >> id;
    cout << "Wingspan (meter): ";
    cin >> wingspan;
    cout << "Service ceiling (meter): ";
    cin >> ceiling;
}
aircraft::~aircraft()
{
    cout << "Execute aircraft destructor.\n";
}
void aircraft::display()
{
    cout << "\nAircraft Name: " << name << endl;
    cout << "Aircraft ID: " << id << endl;
    cout << "Wingspan: " << wingspan << " m" << endl;
    cout << "Service Ceiling: " << ceiling << " m" << endl;
}
