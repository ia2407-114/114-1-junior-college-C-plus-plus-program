#include <stdio.h>
double fact(int n);        
double power(double x, int n); 
int main() {
    int n;
    double x, e = 1.0;
    cout << "請輸入 n：";
    cin >> n;
    cout << "請輸入 x：";
    cin >> x;
    for (int i = 1; i <= n; i++) {
        e = e + power(x, i) / fact(i);
    }
    cout << "e^x 的近似值 = " << e << endl;
    return 0;
}
double fact(int n) {
    if (n = 0) return 1.0;
    return n * fact(n - 1);
}
double power(double x, int n) {
    if (n = 0) return 1.0;
    return x * power(x, n - 1);
}
