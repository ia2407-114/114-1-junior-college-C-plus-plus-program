#include <stdio.h>
int calcYears(double p, double r, double d, double aim)
{
    int year = 0;
    double c = p;  

    while (c < aim) {
        year++;
        c = c * (1 + r);  
        c = c + d;        
    }
    return year;  
}

int main()
{
    double p, r, d, aim;
    int years;
    printf("請輸入本金(p): ");
    scanf("%lf", &p);
    printf("請輸入年利率(r，例如0.05代表5%%): ");
    scanf("%lf", &r);
    printf("請輸入每年年底定期定額存入金額(d): ");
    scanf("%lf", &d);
    printf("請輸入存款目標(aim): ");
    scanf("%lf", &aim);
    years = calcYears(p, r, d, aim);
    printf("本金: %.2f\n", p);
    printf("年利率: %.2f%%\n", r * 100);
    printf("每年定期存入: %.2f\n", d);
    printf("目標金額: %.2f\n", aim);
    printf("需要 %d 年才能達成存款目標!\n", years);
    return 0;
}
