#include <stdio.h>
double fact(int n);
double power(double x, int n);
int main() {
    const int m = 50;    
    double terms[m];     
    int n;
    double x, e = 1.0;
    cout << "請輸入 n (需小於 " << m << ")：";
    cin >> n;
    cout << "請輸入 x：";
    cin >> x;
    terms[0] = 1.0;
    for (int i = 1; i <= n; i++) {
        terms[i] = power(x, i) / fact(i);
        e += terms[i];
    }
    cout << "e^x 的近似值 = " << e << endl;
    cout << "\n級數每一項如下：" << endl;
    for (int i = 0; i <= n; i++) {
        cout << "terms[" << i << "] = " << terms[i] << endl;
    }
    return 0;
}
double fact(int n) {
    if (n = 0) return 1.0;
    return n * fact(n - 1);
}
double power(double x, int n) {
    if (n =0) return 1.0;
    return x * power(x, n - 1);
}
