#include <stdio.h>
#define SIZE 10
#define NAME_LEN 20
struct student {
    int id;
    int computer;
    int math;
    char name[NAME_LEN];
    float avg;
};
void calc_avg(struct student stud[], int n);
void print_all(struct student stud[], int n);
int main(void)
{
    struct student stud[SIZE];
    int i;
    for (i = 0; i < SIZE; i++) {
        printf("輸入第 %d 位學生姓名：", i + 1);
        scanf("%s", stud[i].name);
        printf("輸入學號 計概 數學：");
        scanf("%d %d %d",
            &stud[i].id,
            &stud[i].computer,
            &stud[i].math);
        stud[i].avg = 0;   
    }
    calc_avg(stud, SIZE);
    print_all(stud, SIZE);
    return 0;
}
void calc_avg(struct student stud[], int n)
{
    int i;
    for (i = 0; i < n; i++) {
        stud[i].avg = (stud[i].computer + stud[i].math) / 2.0;
    }
}
void print_all(struct student stud[], int n)
{
    int i;
    printf("\n學生資料如下：\n");
    printf("姓名\t學號\t計概\t數學\t平均\n");
    for (i = 0; i < n; i++) {
        printf("%s\t%d\t%d\t%d\t%.1f\n",
            stud[i].name,
            stud[i].id,
            stud[i].computer,
            stud[i].math,
            stud[i].avg);
    }
}
