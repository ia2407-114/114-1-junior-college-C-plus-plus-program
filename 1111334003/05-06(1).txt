matrix.cpp
#include <iostream>
#include <cmath>
#include "Matrix.h"
using namespace std;
void Matrix::input()
{
    cout << "請輸入矩陣列數與行數: ";
    cin >> row >> col;
    cout << "請輸入矩陣元素:\n";
    for (int i = 0; i < row; i++)
    {
        for (int j = 0; j < col; j++)
        {
            cin >> data[i][j];
        }
    }
}
void Matrix::print()
{
    for (int i = 0; i < row; i++)
    {
        for (int j = 0; j < col; j++)
        {
            cout << data[i][j] << "\t";
        }
        cout << endl;
    }
}
void Matrix::operator!()
{
    for (int i = 0; i < row; i++)
    {
        for (int j = 0; j < col; j++)
        {
            data[i][j] = sqrt(data[i][j]);
        }
    }
}
