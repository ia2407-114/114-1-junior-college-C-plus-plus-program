classroom.h
#ifndef CLASSROOM_H
#define CLASSROOM_H
#include <iostream>
#include <string>
using namespace std;
class student
{
protected:
    int student_id;
    string student_name;
public:
    student(int, string);
};
class teacher
{
protected:
    int teacher_id;
    string teacher_name;
    int course_id;
    string course_name;
    int course_credit;
public:
    teacher(int, string);
    void data_input();
};
class classroom : public student, public teacher
{
public:
    classroom(int, string, int, string);
    void show();
};
#endif#pragma once
