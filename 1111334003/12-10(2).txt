#include <stdio.h>
#define SIZE 20
void bubble(int id[], int score[], int length);
int main(void)
{
    int id[SIZE] = { 12,13,14,15,16,17,19,21,22,23,
                    24,26,27,28,29,31,32,33,36,37 };
    int score[SIZE] = { 66,67,66,70,73,74,75,75,62,32,
                       44,46,27,12,80,63,66,90,11,66 };
    int i;
    puts("原始資料 (學號 / 成績)：");
    for (i = 0; i < SIZE; i++) {
        printf("%4d  %4d\n", id[i], score[i]);
    }
    bubble(id, score, SIZE);
    puts("\n依成績由小到大排序後：");
    for (i = 0; i < SIZE; i++) {
        printf("%4d  %4d\n", id[i], score[i]);
    }
    return 0;
}
void bubble(int id[], int score[], int length)
{
    int pass, i, temp;
    for (pass = 1; pass < length; pass++) {
        for (i = 0; i < length - 1; i++) {
            if (score[i] > score[i + 1]) {
                temp = score[i];
                score[i] = score[i + 1];
                score[i + 1] = temp;
                temp = id[i];
                id[i] = id[i + 1];
                id[i + 1] = temp;
            }
        }
    }
}
