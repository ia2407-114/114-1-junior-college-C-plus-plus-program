matrix.cpp
#include "matrix.h"
Matrix::Matrix(int r, int c)
{
    row = r;
    col = c;
}
void Matrix::input()
{
    cout << "請輸入矩陣元素：" << endl;
    for (int i = 0; i < row; i++)
        for (int j = 0; j < col; j++)
            cin >> data[i][j];
}
void Matrix::print() const
{
    cout << "原始矩陣：" << endl;
    for (int i = 0; i < row; i++) {
        for (int j = 0; j < col; j++)
            cout << data[i][j] << "\t";
        cout << endl;
    }
}
Matrix Matrix::operator-(int b)
{
    Matrix temp(row, col); 
    cout << "矩陣每個元素減 " << b << " 後：" << endl;
    for (int i = 0; i < row; i++) {
        for (int j = 0; j < col; j++) {
            temp.data[i][j] = data[i][j] - b;
            cout << temp.data[i][j] << "\t";
        }
        cout << endl;
    }
    return temp;
}
