#include <stdio.h>

#define SIZE 10
void copy1(char* s1, const char* s2);
void sort_array(char name[][20], int stu[][SIZE], float avg[], int n);

void copy1(char* s1, const char* s2)
{
    for (; (*s1 = *s2) != '\0'; s1++, s2++);
}

void sort_array(char name[][20], int stu[][SIZE], float avg[], int n)
{
    int i, j, temp;
    float tempf;
    char temp_name[20];

    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - 1 - i; j++) {
            if (avg[j] < avg[j + 1]) {

                
                copy1(temp_name, name[j]);
                copy1(name[j], name[j + 1]);
                copy1(name[j + 1], temp_name);

                for (int k = 0; k < 3; k++) {
                    temp = stu[k][j];
                    stu[k][j] = stu[k][j + 1];
                    stu[k][j + 1] = temp;
                }

              
                tempf = avg[j];
                avg[j] = avg[j + 1];
                avg[j + 1] = tempf;
            }
        }
    }
}

int main()
{
    const char* name_ptr[SIZE] = {
        "James","Emily","Michael","Sophia","Daniel",
        "Olivia","William","Emma","Alexar","Chloe"
    };

   
    char name[SIZE][20];

    for (int i = 0; i < SIZE; i++)
        copy1(name[i], name_ptr[i]);


    int stu[3][SIZE] = {
        {1001,1002,1003,1004,1005,1006,1007,1008,1009,1010}, 
        {85,78,90,88,76,92,81,95,84,89},                   
        {80,82,88,90,79,91,85,93,87,86}                    
    };

    float avg[SIZE];

    for (int i = 0; i < SIZE; i++)
        avg[i] = (stu[1][i] + stu[2][i]) / 2.0f;

    printf("排序前：\n");
    printf("姓名\t學號\t計概\t數學\t平均\n");
    for (int i = 0; i < SIZE; i++) {
        printf("%s\t%d\t%d\t%d\t%.1f\n",
            name[i], stu[0][i], stu[1][i], stu[2][i], avg[i]);
    }
    sort_array(name, stu, avg, SIZE);

    printf("\n排序後（依平均成績由大到小）：\n");
    printf("姓名\t學號\t計概\t數學\t平均\n");
    for (int i = 0; i < SIZE; i++) {
        printf("%s\t%d\t%d\t%d\t%.1f\n",
            name[i], stu[0][i], stu[1][i], stu[2][i], avg[i]);
    }

    return 0;
}
