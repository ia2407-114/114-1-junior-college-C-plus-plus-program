#include <stdio.h>
#include <string.h>
#define SIZE 10
struct student{
    char nane[10];
    int id;
    int a;
    int b;
    double avg;
};

void sort_array(struct student mystu[SIZE], int n) {
    int i, j;
    struct student temp;
   
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - 1 - i; j++) {
            if (mystu[j].avg < mystu[j+1].avg) {
                temp = mystu[j];
                mystu[j] = mystu[j+1];
                mystu[j + 1] = temp;
            }
        }
    }
}
int main() {
    struct student mystu[SIZE] = { {"Danny",11,85, 70,},
        {"Mary",15,92,80},
        {"Jimmy",13,78,75,},
        {"Peter",18,85,90},
        {"Sue",12,90,85},
        {"John",17,85,70 },
        {"Hearts",14,76,82},
        {"Diamond",19,95,92},
        {"Clubs",16,88,88},
        {"Spades",20,85,70}};
    int i;

    printf("排序前：\n");
    printf("姓名\t學號\t計概成績\t數學成績\t平均成績\n"); \
        for (i = 0; i < SIZE; i++) {
            mystu[i].avg = (mystu[i].a + mystu[i].b) / 2.0;
        }
    for (i = 0; i < SIZE; i++) {
        printf("%s\t%d\t    %d\t\t    %d\t\t    %f\t\n", mystu[i].nane, mystu[i].id, mystu[i].a, mystu[i].b, mystu[i].avg);
    }
    sort_array(mystu, SIZE);

    printf("\n排序後（依平均成績）：\n");
    printf("姓名\t學號\t計概成績\t數學成績\t平均成績\n"); \

        for (i = 0; i < SIZE; i++) {
            printf("%s\t%d\t    %d\t\t    %d\t\t    %f\t\n", mystu[i].nane, mystu[i].id, mystu[i].a, mystu[i].b, mystu[i].avg);
        }


    return 0;
}
