#include <stdio.h>

double pi(int n);
double pv(double pi, double r);
double pvs(double pi, double r);
double ipv(double pi, double r);
double ipvs(double pi, double r);
double qpv(double pi, double r, double h);
double qpvs(double pi, double r, double h);

int main() {
    int n;
    double r, h;

    printf("輸入項目數項: ");
    scanf_s("%d", &n);
    printf("輸入半徑: ");
    scanf_s("%lf", &r);
    printf("輸入圓柱體的高: ");
    scanf_s("%lf", &h);

    double pi_val = pi(n);

    printf("圓周長是 %.6f\n", pv(pi_val, r));
    printf("圓面積是 %.6f\n", pvs(pi_val, r));
    printf("球體積是 %.6f\n", ipv(pi_val, r));
    printf("球體表面積是 %.6f\n", ipvs(pi_val, r));
    printf("圓柱體積是 %.6f\n", qpv(pi_val, r, h));
    printf("圓柱表面積是 %.6f\n", qpvs(pi_val, r, h));
    printf("pi 值是 %.6f\n", pi_val);

    return 0;
}

double pv(double pi, double r) {
    return 2 * pi * r;
}

double pvs(double pi, double r) {
    return pi * r * r;
}

double ipv(double pi, double r) {
    return (4.0 / 3.0) * pi * r * r * r;
}

double ipvs(double pi, double r) {
    return 4 * pi * r * r;
}

double qpv(double pi, double r, double h) {
    return pi * r * r * h;
}

double qpvs(double pi, double r, double h) {
    return 2 * pi * r * r + 2 * pi * r * h;
}

double pi(int n) {
    double pi_approx = 0.0;
    double sign = 1.0;

    for (int i = 0; i < n; i++) {
        double denominator = 2.0 * i + 1.0;
        pi_approx += sign * (4.0 / denominator);
        sign *= -1.0;
    }

    return pi_approx;
}
