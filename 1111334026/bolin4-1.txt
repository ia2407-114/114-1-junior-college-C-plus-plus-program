#include <stdio.h>

int main(void)
{
	int a, b, c, sum,maxx ,smalle ;
	float avg;
	printf("請輸入第一個整數:");
	scanf_s("%d", &a);
	printf("請輸入第二個整數:");
	scanf_s("%d", &b);
	printf("請輸入第三個整數:");
	scanf_s("%d", &c);
	sum = a + b + c;
	avg = sum / 3.0;
	maxx = a;
	if (b > maxx)
			maxx = b;
	if (c > maxx)
		maxx = c;
	smalle = a;
	if (b < smalle)
		smalle = b;
	if (c < smalle)
		smalle = c;
	printf("和:%d\n", sum);
	printf("平均值:%f\n", avg);
	printf("最大值:%d\n", maxx);
	printf("最小值:%d\n", smalle);
	
	return 0;
}
