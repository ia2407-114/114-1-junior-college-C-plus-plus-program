#include <stdio.h>

int main(void)
{
	float  kgg, mmm, aa;
	printf("請輸入體重(公斤):");
	scanf_s("%f", &kgg);
	printf("請輸入身高(公尺):");
	scanf_s("%f", &mmm);
	aa = kgg / (mmm * mmm);
	printf("你的BMI為:%f\n",aa);
	if (aa < 18.5)
	{
	printf("體重過輕");
	}
	else if (aa <= 24.9)
		{
	    printf("正常體重");
		}
	else if (aa <= 29.9)
			{
	    printf("過重");
			}
	else if (aa <= 34.9)
				{
		printf("輕度肥胖");
				}
	else if (aa <= 39.9)
					{
		printf("中度肥胖");
					}
	else if (aa >= 40)
						{
		printf("重度肥胖(病態肥胖");
						}
						

    return 0;
}
