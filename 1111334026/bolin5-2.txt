#include <stdio.h>

int main(void)

{
	int fact, n;
	printf("請輸入一個整數 fact（< 13）：");
	scanf_s("%d", &fact);
	double fact_result = 1;
	for (int i = 1; i <= fact; i++) {
		fact_result *= i;
	}
	printf("%d! = %f\n", fact, fact_result);
	printf("請輸入計算 e 近似值所需的項目數量 n（< 13）：");
	scanf_s("%d", &n);
	double e = 1.0;
    double factorial = 1;
	for (int i = 1; i <= n; i++) 
	{
		factorial *= i;
		e += 1.0 / factorial;
	}
	printf("e 的近似值（使用 %d 項）：%.10f\n", n, e);
		return 0;
}
