#include <iostream>
#include "volume.h" 

using namespace std;

int main()
{
	
	int i1, i2, i3;
	cout << "請輸入 3 個整數計算體積: ";
	cin >> i1 >> i2 >> i3;
	cout << "整數體積結果為: " << volume(i1, i2, i3) << endl;

	float f1, f2, f3;
	cout << "\n請輸入 3 個浮點數(float)計算體積: ";
	cin >> f1 >> f2 >> f3;
	cout << "浮點數(float)體積結果為: " << volume(f1, f2, f3) << endl;

	double d1, d2, d3;
	cout << "\n請輸入 3 個 double 計算體積: ";
	cin >> d1 >> d2 >> d3;
	cout << "double 體積結果為: " << volume(d1, d2, d3) << endl;

	return 0;
}

template < typename T >
T volume(T side1, T side2, T side3)
{
	return side1 * side2 * side3;
}
