#include <iostream>
#include <string>

using namespace std;

class Pet {
private:
	string name;
	int age;

public:
	Pet(string n, int a) {
		name = n;
		age = a;
		cout << "寵物物件 " << name << " 已建立。" << endl;
	}
	~Pet() {
		cout << "再見了，" << name << "！(解構元已執行)" << endl;
	}
	void displayInfo() {
		cout << "寵物姓名: " << name << ", 年齡: " << age << " 歲" << endl;
	}
};

int main() {
	Pet pet1("小黃", 3);
	Pet pet2("阿福", 5);
	pet1.displayInfo();
	pet2.displayInfo();

	cout << "--- 程式結束前 ---" << endl;

	return 0;
}
