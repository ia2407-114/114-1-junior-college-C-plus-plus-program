#include <iostream>
#include <string>
#include <vector>
#include "pet.h"

using namespace std;

int main() {
    string name, cat;
    int y1, m1, d1, y2, m2, d2;
    vector<Pet*> myPets;

    for (int i = 0; i < 3; ++i) {
        cout << "\n請輸入第 " << i + 1 << " 隻寵物的資料：" << endl;
        cout << "名字: "; cin >> name;
        cout << "品種: "; cin >> cat;

        cout << "請輸入生日 (年 月 日): ";
        cin >> y1 >> m1 >> d1;
        Date birth(y1, m1, d1);

        cout << "請輸入領養日 (年 月 日): ";
        cin >> y2 >> m2 >> d2;
        Date adopt(y2, m2, d2);
        myPets.push_back(new Pet(name, cat, birth, adopt));
    }

    cout << "\n======= 顯示詳細資料與檢查 =======" << endl;
    for (int i = 0; i < 3; ++i) {
        myPets[i]->print_all_data();
        myPets[i]->check_birth();
        myPets[i]->check_adopt();
    }

    cout << "\n======= 程式結束，觸發解構子 =======" << endl;
    for (int i = 0; i < 3; ++i) {
        delete myPets[i];
    }

    return 0;
}

pet.cpp

#include <iostream>
#include "pet.h"
using namespace std;

Pet::Pet(string n, string c, Date b, Date a)
    : name(n), category(c), birthday(b), adoptDate(a) {}

Pet::~Pet() {
    cout << "Pet " << name << " : Bye Bye~~" << endl;
}

void Pet::check_birth() const {
    if (birthday.getMonth() < 1 || birthday.getMonth() > 12) {
        cout << "[警告] " << name << " 的生日月份異常！" << endl;
    }
    else {
        cout << name << " 的生日檢查正常。" << endl;
    }
}

void Pet::check_adopt() const {
    if (adoptDate.getYear() < birthday.getYear()) {
        cout << "[錯誤] " << name << " 的領養年份早於出生年份！" << endl;
    }
    else {
        cout << name << " 的領養日期邏輯正常。" << endl;
    }
}

void Pet::print_all_data() const {
    cout << "---------------------------" << endl;
    cout << "名字: " << name << endl;
    cout << "品種: " << category << endl;
    cout << "生日: "; birthday.printDate(); cout << endl;
    cout << "領養日: "; adoptDate.printDate(); cout << endl;
}

#include <iostream>
#include "date.h"
using namespace std;

Date::Date(int y, int m, int d) : year(y), month(m), day(d) {}

void Date::setDate(int y, int m, int d) {
    year = y; month = m; day = d;
}

void Date::printDate() const {
    cout << year << "/" << month << "/" << day;
}

date.h

#ifndef DATE_H
#define DATE_H

class Date {
private:
    int year, month, day;
public:
    Date(int y = 2000, int m = 1, int d = 1);
    void setDate(int y, int m, int d);
    void printDate() const;
    int getYear() const { return year; }
    int getMonth() const { return month; }
    int getDay() const { return day; }
};
