#include<stdio.h>
double fact(int n);
int main()
{
	int n;
	double e = 1.0;
	printf("請輸入項目數量n:\n");
	scanf_s("%d", &n);

	for (int i = 1; i <= n; i++) {
		e = e + 1.0 / fact(i);
	}
	printf("e的近似值=%1f\n", e);
	return 0;
}
double fact(int n)
{
	if (n == 0) {
		return 1;
	}
	return n * fact(n - 1);
}
