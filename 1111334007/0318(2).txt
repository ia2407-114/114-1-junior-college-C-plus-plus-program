#include <iostream>
using namespace std;

// 1. 正方體
int volume(int s) {
    return s * s * s;
}

// 2. 圓柱體 (PI 取 3.14)
double volume(int r, int h) {
    return 3.14 * r * r * h;
}

// 3. 長方體
int volume(int l, int w, int h) {
    return l * w * h;
}

#include <iostream>
using namespace std;

// 1. 正方體
int volume(int s) {
    return s * s * s;
}

// 2. 圓柱體
double volume(int r, int h) {
    return 3.14 * r * r * h;
}

// 3. 長方體
int volume(int l, int w, int h) {
    return l * w * h;
}

int main() {
    int s, r, h_cyl, l, w, h_rect;

    cout << "=== 請輸入各項參數進行計算 ===" << endl;

    // 輸入正方體參數
    cout << "1. 請輸入正方體邊長: ";
    cin >> s;

    // 輸入圓柱體參數
    cout << "2. 請輸入圓柱體半徑 與 高 (以空格隔開): ";
    cin >> r >> h_cyl;

    // 輸入長方體參數
    cout << "3. 請輸入長方體 長、寬、高 (以空格隔開): ";
    cin >> l >> w >> h_rect;

    cout << "\n--- 計算結果 ---" << endl;

    // 透過傳入不同數量的參數，自動觸發多載函數
    cout << "正方體體積: " << volume(s) << endl;
    cout << "圓柱體體積: " << volume(r, h_cyl) << endl;
    cout << "長方體體積: " << volume(l, w, h_rect) << endl;

    return 0;
}
