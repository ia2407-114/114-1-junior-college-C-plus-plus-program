#include <iostream>
using namespace std;

// 1. 正方體
int volume(int s) {
    return s * s * s;
}

// 2. 圓柱體 (PI 取 3.14)
double volume(int r, int h) {
    return 3.14 * r * r * h;
}

// 3. 長方體
int volume(int l, int w, int h) {
    return l * w * h;
}

int main() {
    int choice;
    cout << "--- 體積計算器 ---" << endl;
    cout << "1. 正方體 (輸入1個參數)" << endl;
    cout << "2. 圓柱體 (輸入2個參數)" << endl;
    cout << "3. 長方體 (輸入3個參數)" << endl;
    cout << "請選擇功能 (1-3): ";
    cin >> choice;

    if (choice == 1) {
        int s;
        cout << "請輸入邊長: ";
        cin >> s;
        cout << "正方體體積為: " << volume(s) << endl;
    }
    else if (choice == 2) {
        int r, h;
        cout << "請輸入底半徑與高 (以空格隔開): ";
        cin >> r >> h;
        cout << "圓柱體體積為: " << volume(r, h) << endl;
    }
    else if (choice == 3) {
        int l, w, h;
        cout << "請輸入長、寬、高 (以空格隔開): ";
        cin >> l >> w >> h;
        cout << "長方體體積為: " << volume(l, w, h) << endl;
    }
    else {
        cout << "輸入錯誤！" << endl;
    }

    return 0;
}
