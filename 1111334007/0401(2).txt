pet.cpp

#include <iostream>
#include "pet.h" 

using namespace std;

Pet::Pet(string n, int a) {
	name = n;
	age = a;
	cout << "寵物物件 " << name << " 已建立。" << endl;
}

Pet::~Pet() {
	cout << "再見了，" << name << "！(解構元已執行)" << endl;
}

void Pet::displayInfo() {
	cout << "寵物姓名: " << name << ", 年齡: " << age << " 歲" << endl;
}
