Dog.cpp

#include <iostream>
#include <string>
#include <stdexcept>
#include "Dog.h"
using namespace std;

Dog::Dog(string n, int by, int bm, int bd, int ay, int am, int ad)
    : name(n), bYear(by), bMonth(bm), bDay(bd), aYear(ay), aMonth(am), aDay(ad) {
    validateStatus(by, bm, bd, ay, am, ad);
}

void Dog::validateStatus(int by, int bm, int bd, int ay, int am, int ad) {

    int currY = 2026, currM = 4, currD = 8;

    int age = currY - by;
    if (currM < bm || (currM == bm && currD < bd)) age--;

    if (age < 0 || age > 20)
        throw invalid_argument("年齡不合理 (需介於 0-20 歲)");

    if (ay < by || (ay == by && am < bm) || (ay == by && am == bm && ad < bd))
        throw invalid_argument("領養日不能早於生日");
}

void Dog::printInfo() const {
    cout << "\n--- 寵物資料 ---"
        << "\n姓名: " << name
        << "\n生日: " << bYear << "/" << bMonth << "/" << bDay
        << "\n領養日: " << aYear << "/" << aMonth << "/" << aDay << endl;
}
