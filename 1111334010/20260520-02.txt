//main.cpp
#include "classroom.h"

int main()
{
    int tid, sid;
    string tname, sname;

    cout << "請輸入教師編號：";
    cin >> tid;

    cout << "請輸入教師姓名：";
    cin >> tname;

    cout << "請輸入學生編號：";
    cin >> sid;

    cout << "請輸入學生姓名：";
    cin >> sname;

    // 建立 classroom 物件
    classroom c1(tid, tname, sid, sname);

    // 輸入課程資料
    c1.data_input();

    // 顯示資料
    c1.show();

    return 0;
}

//classroom.cpp
#include "classroom.h"

// 建構子
classroom::classroom(int tid, string tname,
                     int sid, string sname)
    : teacher(tid, tname),
      student(sid, sname)
{
}

// 顯示資料
void classroom::show()
{
    cout << "\n===== classroom 資料 =====" << endl;

    cout << "課程編號：" << course_id << endl;
    cout << "課程名稱：" << course_name << endl;
    cout << "課程學分：" << course_credit << endl;

    cout << "教師姓名：" << teacher::name << endl;
    cout << "學生姓名：" << student::name << endl;
}

//classroom.h
#ifndef CLASSROOM_H
#define CLASSROOM_H

#include "teacher.h"
#include "student.h"

// 多重繼承
class classroom : public teacher, public student
{
public:
    // 建構子
    classroom(int, string, int, string);

    // 顯示所有資料
    void show();
};

#endif
