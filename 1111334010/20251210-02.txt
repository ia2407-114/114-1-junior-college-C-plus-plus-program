#include <stdio.h>

int main()
{
    int id[20] = { 12,13,14,15,16,17,19,21,22,23,
                  24,26,27,28,29,31,32,33,36,37 };

    int score[20] = { 66,67,66,70,73,74,75,75,62,32,
                     44,46,27,12,80,63,66,90,11,66 };

    int i, j;
    int temp;

    /* 氣泡排序（依成績由小到大） */
    for (i = 0; i < 19; i++)
    {
        for (j = 0; j < 19 - i; j++)
        {
            if (score[j] > score[j + 1])
            {
                /* 交換成績 */
                temp = score[j];
                score[j] = score[j + 1];
                score[j + 1] = temp;

                /* 同步交換學號 */
                temp = id[j];
                id[j] = id[j + 1];
                id[j + 1] = temp;
            }
        }
    }

    /* 印出排序後結果 */
    printf("排序後結果（學號 / 成績）：\n");
    for (i = 0; i < 20; i++)
    {
        printf("學號：%d 成績：%d\n", id[i], score[i]);
    }

    return 0;
}
