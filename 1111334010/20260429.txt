//count.h
#ifndef COUNT_H
#define COUNT_H

#include <iostream>
using namespace std;

class printDate; // 前向宣告

class Count
{
    friend void setX(Count&, int);
    friend class printDate;

public:
    Count();
    void print() const;

private:
    int x;
};

void setX(Count&, int);

class printDate
{
public:
    void showX(const Count& c);
};

#endif

//count.cpp
#include "Count.h"

Count::Count() : x(0)
{
}

void Count::print() const
{
    cout << x << endl;
}

void setX(Count& c, int val)
{
    c.x = val;
}

void printDate::showX(const Count& c)
{
    cout << "printDate 類別輸出 x: " << c.x << endl;
}

//main.cpp
#include <iostream>
#include "Count.h"

using namespace std;

int main()
{
    Count counter;
    printDate pd;

    setX(counter, 8);

    pd.showX(counter); // 輸出 8

    return 0;
}
