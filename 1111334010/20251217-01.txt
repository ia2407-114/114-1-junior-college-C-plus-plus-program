#include <stdio.h>
#define SIZE 10

void sort_array(int id[], int computer[], int math[], float avg[], int n) {
    int i, j, temp;
    float temp1;

    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - 1 - i; j++) {
            if (avg[j] > avg[j + 1]) {

                temp = id[j];
                id[j] = id[j + 1];
                id[j + 1] = temp;

                temp = computer[j];
                computer[j] = computer[j + 1];
                computer[j + 1] = temp;

                temp = math[j];
                math[j] = math[j + 1];
                math[j + 1] = temp;

                temp1 = avg[j];
                avg[j] = avg[j + 1];
                avg[j + 1] = temp1;

            }
        }
    }
}


int find_mid(float arr[], int n) {


    return arr[n / 2];
}

    }
