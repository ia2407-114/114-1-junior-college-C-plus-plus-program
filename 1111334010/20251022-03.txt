#include<stdio.h>

int main(void)
{
	double p;
	double r;
	double month;
	double aims;
	double total;
	int year = 0;

	printf("請輸入本金:\t");
	scanf_s("%lf", &p);
	printf("請輸入年利率:\t");
	scanf_s("%lf", &r);
	printf("請輸入每月存入金額:\t");
	scanf_s("%lf", &month);
	printf("請輸入存款目標:\t");
	scanf_s("%lf", &aims);

	total = p;

	while (total < aims) {
		year++;
		total += month * 12;
		total = total * (1 + r);
		printf("第%d年底，本利合為:%.2f\n",year,total);
	}
	printf("共需%d年才能達成存款目標!",year);

	return 0;
}
