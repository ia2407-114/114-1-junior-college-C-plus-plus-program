#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

struct student {
    int id;
    char name[20];
    int computer;
    int math;
    double avg;
};

int main(void)
{
    FILE* cfPtr;

    if ((cfPtr = fopen("student.dat", "wb")) == NULL) {
        puts("File could not be opened.");
    }
    else {

        struct student s;

        printf("ID    Name                 Computer   Math   Avg\n");

        while (fread(&s, sizeof(struct student), 1, cfPtr) == 1) {

            if (s.id != 0) {
                printf("%-6d%-20s%-10d%-10d%-10.2f\n",
                    s.id, s.name, s.computer, s.math, s.avg);
            }
        }

        fclose(cfPtr);
    }

    return 0;
}
