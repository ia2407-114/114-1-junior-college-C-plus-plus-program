#include<stdio.h>
#include<stdlib.h>

int main()
{
	int i = 0;
	float p;
	float sum;
	float r;
	float aims;
	printf("請輸入本金\n");
	scanf_s("%f", &p);
	printf("請輸入年利率\n");
	scanf_s("%f", &r);
	printf("請輸入存款目標\n");
	scanf_s("%f", &aims);
	sum = p;
	while (sum < aims) {
		i++;
		sum = sum*(1 + r);
		printf("第%d年本利合: %.2f\n", i, sum);
	}
	printf("預計需%d年來達成目標",i);
return 0;
}
