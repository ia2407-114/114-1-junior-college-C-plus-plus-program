#include <stdio.h>
#include <stdlib.h>

int main() {
	float height;
	float weight;
	float BMI;
	printf("請注意身高為公尺,體重為KG!\n");
	printf("請輸入您的身高及體重!\n");
	scanf_s("%f%f", &height, &weight);
	float cm = height * height;
	BMI = (float)(weight / cm);
	printf("你的BMI數值為:%f",BMI);
	if (BMI < 18.5)
	{
		printf("你的BMI等級為:體重過輕\n");
	}
	else if (BMI <=24.9)
	{
		printf("你的BMI等級為:正常體重\n");

	}
	else if (BMI <=29.9)
	{
		printf("你的BMI等級為:過重\n");

	}
	else if (BMI <= 34.9)
	{
		printf("你的BMI等級為:輕度肥胖\n");

	}
	else if (BMI <= 39.9)
	{
		printf("你的BMI等級為:中度肥胖\n");

	}
	else if (BMI >= 40.0)
	{
		printf("你的BMI等級為:重度肥胖\n");

	}
	printf("系統執行完畢");
	
		return 0;
}
