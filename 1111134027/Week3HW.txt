#include<stdio.h>
#include <stdlib.h>

int main() {
	int chinese, math, itc;
	float avg;
	 printf("請輸入國文,數學,計概成績");
	 scanf_s("%d%d%d",&chinese,&math,&itc);
	 printf("總成績:%d\n",chinese + math + itc);
	 avg =(float)(chinese + math + itc) / 3;
	 printf("平均成績:%f\n",avg);
	if (avg >= 60) {
		printf("恭喜通過!");
	}
	else {
		printf("再加油!");

	}
    return 0;
}
