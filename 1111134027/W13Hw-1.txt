#include <stdio.h>
#define SIZE 5       
#define TOTAL 20      

int main(void)
{
    int n[SIZE] = { 0 };       
    int input;

   
    printf("請輸入 20 份問卷結果 (1~5)：\n");
    for (int i = 0; i < TOTAL; i++) {
        printf("第 %d 份：", i + 1);
        scanf_s("%d", &input);

      
        if (input >= 1 && input <= 5) {
            n[input - 1]++;  
        }
        else {
            printf("無效輸入，請重新輸入！\n");
            i--; 
        }
    }

    printf("\n%s%13s%17s\n", "選項(1-5)", "次數", "Histogram");

  
    for (size_t i = 0; i < SIZE; ++i) {
        printf("%7u%13d        ", i + 1, n[i]);

        for (int j = 1; j <= n[i]; ++j) {
            printf("%c", '*');
        }

        puts("");
    }

    return 0;
}
