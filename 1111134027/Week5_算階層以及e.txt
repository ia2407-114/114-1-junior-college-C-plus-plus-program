#include <stdio.h>
#include <stdlib.h>

int main() {
	int fact, n;
	int i, j;
	long factorial;
	double e = 1.0;

	printf("請輸入一個整數 fact (<13):\n");
	scanf_s("%d", &fact);
	factorial = 1;

	for (i = 1; i <= fact; i++) {
		factorial *= i;
	}
	printf("%d! = %1d\n",fact,factorial);
	printf("請輸入項目數量 n (<13):");
	scanf_s("%d", &n);
	
	for (i = 1; i <= n; i++) {
		long term_fact = 1;
		for (j = 1; j <= i; j++) {
			term_fact *= j;
		}
		e += 1.0 / term_fact;
	}
	printf("e的近似值(取前%d項 = %.10f\n", n, e);




	return 0;
}
