#include <stdio.h>
int main(void)
{
	float kg, m, bmi;
	      printf("請輸入你的體重 \n");
	      scanf_s("%f", &kg);
	      printf("請輸入你的身高(公尺)\n");
	      scanf_s("%f", &m);
       

	bmi = kg / (m*m) ;
	   printf("你的BMI是%f",bmi);
	 if (bmi < 18.5)
		    printf("你的體重過輕");
	 else if (bmi <= 24.9)
		    printf("你的正常體重");
	 else if (bmi <= 29.9)
		    printf("你的過重");
	 else if (bmi <= 34.9)
		    printf("你的輕度肥胖");
	 else if (bmi <= 39.9)
		    printf("你的中度肥胖");
	 else if (bmi >= 40)
		    printf("你的重度肥胖");
