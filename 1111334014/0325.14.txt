#include <iostream> #include <string> using namespace std; // Dog class definition class Dog { public: // function that sets the dog name void setName(string n) { if (n.length() > 2 && n.length() < 5) name = n; else cout << "Name length error!" << endl; } // function that sets the dog age void setAge(int a) { if (a <= 20) age = a; else cout << "Age error!" << endl; } // function that gets and displays dog info void getInfo() const { cout << "Dog name: " << name << endl; cout << "Dog age: " << age << endl; } private: string name; int age; }; // main function int main() { string dogName; int dogAge; Dog myDog; cout << "Please enter dog name:" << endl; getline(cin, dogName); myDog.setName(dogName); cout << "Please enter dog age:" << endl; cin >> dogAge; myDog.setAge(dogAge); cout << endl; myDog.getInfo(); return 0; }
