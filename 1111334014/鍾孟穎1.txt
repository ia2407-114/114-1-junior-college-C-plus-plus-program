#include <stdio.h>

int main() {
int a, b, c;
int max, min;
int sum;
float avg;

printf("請輸入三個整數：");
scanf("%d %d %d", &a, &b, &c);

// 計算總和與平均
sum = a + b + c;
avg = sum / 3.0;

// 找最大值（只能用單一條件）
max = a;
if (b > max) max = b;
if (c > max) max = c;

// 找最小值（只能用單一條件）
min = a;
if (b < min) min = b;
if (c < min) min = c;

// 輸出結果
printf("總和 = %d\n", sum);
printf("平均值 = %.2f\n", avg);
printf("最大值 = %d\n", max);
printf("最小值 = %d\n", min);

return 0;
}
