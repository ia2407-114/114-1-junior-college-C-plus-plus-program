#include <stdio.h>

int main() {
    float weight, height, bmi;

    printf("請輸入體重(公斤): ");
    scanf_s("%f", &weight);

    printf("請輸入身高(公尺): ");
    scanf_s("%f", &height);

    // 計算 BMI
    bmi = weight / (height * height);

    printf("您的BMI為: %.2f\n", bmi);

    // 顯示等級（每個 if 只使用單一條件式）

    if (bmi < 18.5) {
        printf("體重過輕\n");
    }
    else if (bmi < 25.0) {
        printf("正常體重\n");
    }
    else if (bmi < 30.0) {
        printf("過重\n");
    }
    else if (bmi < 35.0) {
        printf("輕度肥胖\n");
    }
    else if (bmi < 40.0) {
        printf("中度肥胖\n");
    }
    else
        printf("重度肥胖（病態肥胖）\n");
    
    
    return 0;
}
