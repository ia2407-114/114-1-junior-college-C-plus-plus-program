#include <stdio.h>
int main()
{
	int c = 10000, aims, add = 0, years = 0;
	float r = 1.05;
	printf("本金 10000, 利率5 percent, 請輸入你的目標金額\n");
	scanf_s("%d", &aims);
	printf("請輸入每一年底要再存多少錢進戶頭\n");
	scanf_s("%d", &add);
	while (c < aims)
	{
		years += 1;
		c = c * r;
		c = c + add;
		printf("第%d年存了%d元\n", years, c);
	}
	printf("共存了%d元，需要%d年", c, years);
}
