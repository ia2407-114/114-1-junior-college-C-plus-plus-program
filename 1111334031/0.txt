#include <stdio.h>

int main(void)
{
    int chinese, english, planoutline, total;
    float avg;
    printf("請輸入你的國文成績:\n");
    scanf_s("%d", &chinese);
    printf("請輸入你的英文成績:\n");
    scanf_s("%d", &english);
    printf("請輸入你的機概成績:\n");
    scanf_s("%d", &planoutline);
    total = chinese + english + planoutline;
    printf("你的總成績%d+%d+%d是:%d\n", chinese, english, planoutline, total);
    avg = total / 3.0;
    printf("你的平均績是:%f\n", avg);

    if (avg >= 60)
    {
        printf("恭喜通過!");
    }
    else
    {
        printf("再加油!");
    }
}
