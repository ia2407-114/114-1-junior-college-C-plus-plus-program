#include <stdio.h>

int main(void)
{
    int a, b, c;
    int sum;
    double avg;
    int max, min;

    printf("請輸入三個整數:\n");
    scanf_s("%d %d %d", &a, &b, &c);

    sum = a + b + c;
    avg = sum / 3.0;

    max = a;
    min = a;

    if (b > max)
        max = b;
    if (c > max)
        max = c;

    if (b < min)
        min = b;
    if (c < min)
        min = c;

    printf("三個整數的和 = %d\n", sum);
    printf("平均值 = %.2f\n", avg);
    printf("最大值 = %d\n", max);
    printf("最小值 = %d\n", min);

    return 0;
}
