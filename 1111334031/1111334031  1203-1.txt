#include <stdio.h>

int factorial(int n) {
    int result = 1;
    for (int i = 1; i <= n; i++) {
        result *= i;
    }
    return result;
}

int main() {
    int n;
    printf(“輸入n: “);
    scanf_s(“%d”, &n);

    double e = 1.0;
    for (int i = 1; i <= n; i++) {
        e += 1.0 / factorial(i);
    }

    printf(“e = %lf\n”, e);
    return 0;
}
