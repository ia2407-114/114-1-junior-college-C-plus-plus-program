#include <stdio.h>
int main()
{
    //題目1: 從鍵盤輸入本金(p=10000) 年利率(r=0.05)，以及存款目標(aims)印出每年底(複利)本利合 直到達成存款目標，讓使用者知道幾年才能達成目標
        double p, r, aims;
        int year = 0;

        // 輸入本金、年利率、目標金額
        printf("請輸入本金 (例如 10000): ");
        scanf("%lf", &p);

        printf("請輸入年利率 (例如 0.05 代表5%%): ");
        scanf_s("%lf", &r);

        printf("請輸入存款目標金額: ");
        scanf_s("%lf", &aims);

        double amount = p;

        // 顯示表頭
        printf("\n年數\t本利合計\n");

        // 每年複利直到達成目標
        while (amount < aims) {
            year++;
            amount = amount * (1 + r);
            printf("%d\t%.2f\n", year, amount);
        }

        // 顯示總年數
        printf("\n達成目標需時 %d 年。\n", year);

        return 0;
}
