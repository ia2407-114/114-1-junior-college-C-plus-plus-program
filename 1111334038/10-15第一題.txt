#include<stdio.h>
int main()
{
    int kg , price=60,weight=2;
    printf("請輸入包裹重量:\n");
    scanf_s("%d", &kg);
    if (kg < 2)
        printf("包裹太輕了");
    while (weight<kg and kg>2)
    {
        weight = weight + 1, price = price + 35;
    }
    printf("你的物流費用是%d元", price);
    return 0;
    
    

}
