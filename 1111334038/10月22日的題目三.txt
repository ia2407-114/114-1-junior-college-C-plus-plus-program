#include<stdio.h>
int main()
{
    //題目 3: 從鍵盤輸入本金(p) 年利率(r)，每個"月"底定期定額存入金額，以及存款目標(aims)印出每年底(複利)本利合 直到達成存款目標，讓使用者知道幾年才能達成目標

    double p;         // 本金（使用者一次性投入的初始金額）
    double r;         // 年利率（例如 0.05 表示 5%）
    double deposit;   // 每年底固定定期定額存入的金額
    double aims;      // 存款目標金額（想要達到的金額）
    double amount;    // 總資產金額（本金 + 每年利息 + 每年存入的金額）
    int year = 0;     // 經過的年數（用來計算達成目標所需時間）

    // === 使用者輸入本金、利率、每年定存金額、目標金額 ===
    printf("請輸入本金（例如 10000）：");
    scanf_s("%lf", &p);

    printf("請輸入年利率（例如 0.05 代表5%%）：");
    scanf_s("%lf", &r);

    printf("請輸入每年底定期定額存入金額：");
    scanf_s("%lf", &deposit);

    printf("請輸入目標金額：");
    scanf_s("%lf", &aims);

    // 將本金指定給資產總額（初始值）
    amount = p;

    // 印出表格標題
    printf("\n年數\t本利合計\n");

    // === 開始進入每年累積的複利計算迴圈 ===
    while (amount < aims) {
        year++;                       // 年份 +1
        amount = amount * (1 + r);   // 本利和先乘上年利率（複利成長）
        amount += deposit;           // 每年年底存入固定金額
        printf("%d\t%.2f\n", year, amount); // 印出目前年數與總金額
    }

    // === 印出達成目標所需的總年數 ===
    printf("\n達成目標需時 %d 年。\n", year);

    return 0;
}
