實作檔:

#include <iostream>
#include <iomanip>
#include <stdexcept>
#include "dog.h"
using namespace std;


Dog::Dog(string n, int by, int bm, int bd, int ay, int am, int ad)
{
    name = n;
    byear = by;
    bmonth = bm;
    bday = bd;
    ayear = ay;
    amonth = am;
    aday = ad;
}
bool Dog::isValid()
{
    int currentYear = 2026;

    int age = currentYear - byear;

    if (age < 0 || age > 20)
        return false;

    if (ayear < byear)
        return false;

    return true;
}

void Dog::showInfo()
{
    cout << "姓名: " << name << endl;
    cout << "生日: " << byear << "/" << bmonth << "/" << bday << endl;
    cout << "領養日: " << ayear << "/" << amonth << "/" << aday << endl;
}
