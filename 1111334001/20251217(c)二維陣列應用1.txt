#include <stdio.h>
#define SIZE 10

void sort_array(int s[][10], float avg[], int n) {
    int i, j, temp;
    float temp1;

    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - 1 - i; j++) {
            if (avg[j] > avg[j + 1]) {

                temp = s[0][j];
                s[0][j] = s[0][j + 1];
                s[0][j + 1] = temp;

                temp = s[1][j];
                s[1][j] = s[1][j + 1];
                s[1][j + 1] = temp;

                temp = s[2][j];
                s[2][j] = s[2][j + 1];
                s[2][j + 1] = temp;

                temp1 = avg[j];
                avg[j] = avg[j + 1];
                avg[j + 1] = temp1;

            }
        }
    }
}


int find_mid(float arr[], int n) {


    return arr[n / 2];
}



int main() {
    int s[3][10] = { { 11, 15, 13, 18, 12, 17, 14, 19, 16, 20 },{ 85, 92, 78, 85, 90, 85, 76, 95, 88, 85 },{ 70, 80, 75, 90, 85, 70, 82, 92, 88, 70 } };

    float average_scores[SIZE];

    int i;

    for (i = 0; i < SIZE; i++) {
        average_scores[i] = (float)(s[1][i] + s[2][i]) / 2.0;
    }

    printf("學號\t計概成績\t數學成績\t平均成績\n");
    for (i = 0; i < SIZE; i++) {
        printf("%d\t    %d\t\t    %d\t  %f\t\t\n", s[0][i], s[1][i], s[2][i], average_scores[i]);
    }
    printf("\n");


    sort_array(s, average_scores, SIZE);
    float avg_mode = find_mid(average_scores, SIZE);


    printf("學號\t計概成績\t數學成績\t平均成績\n");
    for (i = 0; i < SIZE; i++) {
        printf("%d\t    %d\t\t    %d\t  %f\t\t\n", s[0][i], s[1][i], s[2][i], average_scores[i]);
    }
    printf("\n");
    printf("\n\n");

    printf("平均中位數 ：%f\n", avg_mode);

    return 0;
}
