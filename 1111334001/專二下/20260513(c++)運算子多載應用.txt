資源檔(Date.cpp)
#include <iostream>
#include <string>
#include "Date.h"
using namespace std;


Date::Date(int hour, int minute, int second)
{
    setDate(hour, minute, second);
} 


void Date::setDate(int hh, int mm, int ss)
{
    if (hh >= 0 && hh <= 23)
        hour = hh;
    else
        throw invalid_argument("Hour must be 0-23");

    if (mm >= 0 && mm <= 59)
        minute = mm;
    else
        throw invalid_argument("Minute must be 0-59");

    if (ss >= 0 && ss <= 59)
        second = ss;
    else
        throw invalid_argument("Second must be 0-59");
    
} 
Date& Date::operator++()
{
    helpIncrement(); 
    return *this; 
} 
Date Date::operator++(int)
{
    Date temp = *this; 
    helpIncrement();

    
    return temp; 
} 
Date& Date::operator+=(unsigned int additionalDays)
{
    for (int i = 0; i < additionalDays; ++i)
        helpIncrement();

    return *this; 
} 


void Date::helpIncrement()
{
    
    if (second<59)
        ++second;
    else
        if (minute < 59)
        {
            ++minute;
            second = 0;
        } 
        else 
            if (hour < 23)
        {
            ++hour; 
            minute = 0; 
            second = 0; 
        }
            else {
                hour=0;
                minute = 0;
                second = 0;
            }
} 

ostream& operator<<(ostream& output, const Date& d)
{

    output << d.hour << ":" << d.minute << ":" << d.second;
    return output;
}
