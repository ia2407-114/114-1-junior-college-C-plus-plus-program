#include <iostream>
using namespace std;
double power(int&, int&);
double fact(int&);

int  main(void){
	
		int x, z, i, k, j;   //ex   e^1=2.71828…
		cout << "請輸入一個X值 =";
		cin >> x ;
		cout << "請輸入階乘值 =";
		cin >> i;
		cout << "階乘答案為:" << power(x, i) << "，次方答案為:" << fact(i);
		return 0;
	}

	double fact(int& i)
	{
		int j;
		double level = 1.0;
		for (j = i; j >= 1; j--) {//階乘 e.g. i=5
			level *= j;  // level=level * j  ?5 * 4 * 3 * 2 * 1
		}  // 5!=120  
		return level ;
	}

	double power(int& x, int& y)
	{
		double xpow = 1;
		int k;
		for (k = 1; k <= y; k++) {//次方  e.g. i=5
			xpow *= x;         // xpow=xpow * x ?5 * 5 * 5 * 5 * 5 * 5= 3125
		}
		return xpow;
	}
