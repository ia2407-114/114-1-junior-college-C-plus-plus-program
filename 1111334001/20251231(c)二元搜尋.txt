#include <stdio.h>
#include <string.h>
#define SIZE 10

int binarySearch(int stu[], int searchKey, int low, int high)
{

    while (low <= high) {

        int middle = (low + high) / 2;


        if (searchKey == stu[middle]) {  
            return middle;
        }


        else if (searchKey < stu[middle]) {
            high = middle - 1;
        }
        else {   
            low = middle + 1;
        }
    } 

    return -1; 
}


void copy1(char* s1, const char* s2)
{
    for (; ((*s1 = *s2) != '\0'); ++s1, ++s2);
}
void swap(int* prt1, int* prt2) {
    int temp = *prt1;
    *prt1 = *prt2;
    *prt2 = temp;
}
void swap_f(float* prt1, float* prt2) {
    float temp = *prt1;
    *prt1 = *prt2;
    *prt2 = temp;
}
void sort_array(char name[][20], int stu[][SIZE], float avg[], int n) {
    int i, j, temp;
    float temp1;
    char temp_name[20];


    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - 1 - i; j++) {
            if (avg[j] < avg[j + 1]) {

                copy1(temp_name, name[j]);
                copy1(name[j], name[j + 1]);
                copy1(name[j + 1], temp_name);
                swap(&stu[0][j], &stu[0][j + 1]);
                swap(&stu[1][j], &stu[1][j + 1]);
                swap(&stu[2][j], &stu[2][j + 1]);
                swap_f(&avg[j], &avg[j + 1]);

            }
        }
    }
}




int main() {
    const char* stu_name[SIZE] = { "Danny","Mary","Jimmy","Peter","Sue","John","Hearts", "Dia","Clubs","Spades" };
    char stu1_name[10][20];

    for (int i = 0; i < SIZE; i++)
        copy1(stu1_name[i], stu_name[i]);

    int stu[3][SIZE] = { { 11, 15, 23, 33, 42, 46, 57, 79, 85, 95 },
                         { 85, 92, 78, 85, 90, 85, 76, 95, 88, 85 },
                         { 70, 80, 75, 90, 85, 70, 82, 92, 88, 70 } };

    float average_scores[SIZE];

    int i, key, result;
    int low = 0;
    printf("請輸入你要找的學號鍵值:");
    scanf_s("%d", &key);
    result = binarySearch(stu[0], key, low, SIZE - 1);
    if (result == -1) {
        printf("找不到資料\n");
    }
    else { 
        printf("這位學號是%d的同學的名字%s,%d,%d,%f\n", key, stu_name[result], stu[1][result], stu[2][result], (stu[1][result] + stu[2][result]) / 2.);//因為平均下面才開始算所以先直接硬算
    }
    printf("排序前：\n");
    printf("姓名\t學號\t計概成績\t數學成績\t平均成績\n"); \
        for (i = 0; i < SIZE; i++) {
            average_scores[i] = (stu[1][i] + stu[2][i]) / 2.0;
        }
    for (i = 0; i < SIZE; i++) {
        printf("%s\t%d\t    %d\t\t    %d\t\t    %f\t\n", stu1_name[i], stu[0][i], stu[1][i], stu[2][i], average_scores[i]);
    }


    sort_array(stu1_name, stu, average_scores, SIZE);

    printf("\n排序後（依平均成績）：\n");
    printf("姓名\t學號\t計概成績\t數學成績\t平均成績\n"); \

        for (i = 0; i < SIZE; i++) {
            printf("%s\t%d\t    %d\t\t    %d\t\t    %f\t\n", stu1_name[i], stu[0][i], stu[1][i], stu[2][i], average_scores[i]);
        }


    return 0;
}
