#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

struct student {
    int id;
    char name[20];
    int computer, math;
    double avg;
};

int main(void)
{
    FILE* cfPtr;

    if ((cfPtr = fopen("student.dat", "wb")) == NULL) {
        puts("File could not be opened.");
        return 1;
    }

    printf("%-6s%-20s%-10s%-10s%-10s\n",
        "ID", "Name", "Computer", "Math", "Average");

    struct student stu;

    while (fread(&stu, sizeof(struct student), 1, cfPtr) == 1) {


        if (stu.id != 0) {
            printf("%-6d%-20s%-10d%-10d%-10.2f\n",
                stu.id,
                stu.name,
                stu.computer,
                stu.math,
                stu.avg);
        }
    }

    fclose(cfPtr);

    return 0;
}
