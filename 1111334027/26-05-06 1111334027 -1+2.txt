#include <iostream>
#include "Matrix.h"
using namespace std;

int main()
{
    int i, j, row, column;

    cout << "測試 !（開根號）與 ++（平方）運算子\n";
    cout << "輸入列數(row): ";
    cin >> row;
    cout << "輸入行數(column): ";
    cin >> column;

    Matrix a(row, column), b(row, column), c(row, column);

    cout << "輸入矩陣 A：" << endl;
    for (i = 0; i < row; i++)
        for (j = 0; j < column; j++)
        {
            cout << "A[" << i << "][" << j << "] = ";
            cin >> a.element[i][j];
        }

    b = !a;  
   

    cout << "\n!A（平方根）：\n";
    for (i = 0; i < row; i++)
    {
        for (j = 0; j < column; j++)
            cout << b.element[i][j] << "\t";
        cout << endl;
    }

    c = a++;
    cout << "\na++（平方）：\n";
    for (i = 0; i < row; i++)
    {
        for (j = 0; j < column; j++)
            cout << c.element[i][j] << "\t";
        cout << endl;
    }

    return 0;
}

Matrix.cpp
#include "Matrix.h"
#include <cmath>


Matrix::Matrix(int m, int n)
{
    row = m;
    column = n;

    element = new double* [row];
    for (int i = 0; i < row; i++)
        element[i] = new double[column];
}

Matrix Matrix::operator!()
{
    Matrix temp(row, column);

    for (int i = 0; i < row; i++)
        for (int j = 0; j < column; j++)
        {
            if (element[i][j] >= 0)
                temp.element[i][j] = sqrt(element[i][j]);//sqrt是開根號
            else
                temp.element[i][j] = 0;
        }

    return temp;
}


Matrix Matrix::operator++(int)
{
    Matrix temp(row, column);

    for (int i = 0; i < row; i++)
        for (int j = 0; j < column; j++)
        {
            temp.element[i][j] = element[i][j] * element[i][j];
        }

    return temp;
}
