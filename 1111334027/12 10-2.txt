#include <stdio.h>
#define SIZE 20
void sort_id_score(int id_arr[], int score_arr[], int length);

int main(void)
{
    int id[SIZE] = { 12, 13, 14, 15, 16, 17, 19, 21, 22, 23, 24, 26, 27, 28, 29, 31, 32, 33, 36, 37 };
    int score[SIZE] = { 66, 67, 66, 70, 73, 74, 75, 75, 62, 32, 44, 46, 27, 12, 80, 63, 66, 90, 11, 66 };
    puts("原始資料");
    printf("%-8s %-8s\n", "學號", "成績");
    for (size_t i = 0; i < SIZE; ++i) 
    {
        printf("%-8d %-8d\n", id[i], score[i]);
    }
    sort_id_score(id, score, SIZE);
    puts("\n排序後結果 (成績由小到大)");
    printf("%-8s %-8s\n", "學號", "成績");
    for (size_t i = 0; i < SIZE; ++i) 
    {
        printf("%-8d %-8d\n", id[i], score[i]);
    }
    puts("");
    return 0;
}

void sort_id_score(int id_arr[], int score_arr[], int length)
{
    int pass, i;
    int temp_score;
    int temp_id;

    for (pass = 1; pass < length; ++pass)
    {
        for (i = 0; i < length - 1; ++i) 
        {
            if (score_arr[i] > score_arr[i + 1]) 
            {
                temp_score = score_arr[i];
                score_arr[i] = score_arr[i + 1];
                score_arr[i + 1] = temp_score;

                temp_id = id_arr[i];
                id_arr[i] = id_arr[i + 1];
                id_arr[i + 1] = temp_id;
            }
        }
    }
}
