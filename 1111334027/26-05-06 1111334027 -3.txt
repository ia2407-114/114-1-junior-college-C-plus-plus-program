#include <iostream>
#include "Matrix.h"
using namespace std;
 
int main()
{
    int i, j, row, column, b;

    cout << "輸入矩陣列數(row): ";
    cin >> row;
    cout << "輸入矩陣行數(column): ";
    cin >> column;

    Matrix a('A', row, column);
    Matrix c(row, column);


    cout << "\n原始矩陣 A = \n";
    for (i = 0; i < row; i++)
    {
        for (j = 0; j < column; j++)
            cout << a.element[i][j] << '\t';
        cout << endl;
    }

    cout << "\n輸入整數 b: ";
    cin >> b;

    c = a - b;


    cout << "\n矩陣 A - " << b << " = \n";
    for (i = 0; i < row; i++)
    {
        for (j = 0; j < column; j++)
            cout << c.element[i][j] << '\t';
        cout << endl;
    }

    return 0;
}

Matrix.cpp
#include "Matrix.h"


Matrix::Matrix(int m, int n)
{
    row = m;
    column = n;

    element = new int* [m];
    for (int i = 0; i < m; i++)
        element[i] = new int[n];
}


Matrix::Matrix(char name, int m, int n)
{
    row = m;
    column = n;

    element = new int* [m];
    for (int i = 0; i < m; i++)
        element[i] = new int[n];

    cout << "輸入一 " << m << "x" << n << " 矩陣 " << name << endl;

    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
        {
            cout << name << "[" << i << "][" << j << "] = ";
            cin >> element[i][j];
        }
}

Matrix Matrix::operator-(int b)
{
    Matrix temp(row, column);

    for (int i = 0; i < row; i++)
        for (int j = 0; j < column; j++)
            temp.element[i][j] = element[i][j] - b;

    return temp;
}

void Matrix::print()
{
    for (int i = 0; i < row; i++)
    {
        for (int j = 0; j < column; j++)
            cout << element[i][j] << "\t";
        cout << endl;
    }
}
