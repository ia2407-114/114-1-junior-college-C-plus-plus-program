#include <stdio.h>
double factorial(int fact);
int main() 
{
    int n;
    double result;
    printf("請輸入一個整數：");
    scanf("%d", &n);
    result = factorial(n);
    printf("%d 的階乘為：%.0lf\n", n, result);
    return 0;
}

double factorial(int fact)
{
    double ans = 1; 
    for (int i = 1; i <= fact; i++)
    {
        ans = ans * i;
    }
    return ans;
}
