#include <stdio.h>
#include <stdlib.h>

struct student {
    int id;
    char name[20];
    int computer;
    int math;
    double avg;
};

int main() {

    FILE* fp = NULL;
    struct student s;

    fopen_s(&fp, "student.dat", "wb+");

    if (fp == NULL) {
        printf("檔案開啟失敗\n");
        return 1;
    }

    printf("請輸入學生資料 (輸入 id=0 結束)\n");

    while (1) {

        printf("輸入學號: ");
        scanf_s("%d", &s.id);

        if (s.id == 0)
            break;

        printf("輸入姓名: ");
        scanf_s("%19s", s.name, (unsigned)_countof(s.name));

        printf("輸入計概成績: ");
        scanf_s("%d", &s.computer);

        printf("輸入數學成績: ");
        scanf_s("%d", &s.math);

        s.avg = (s.computer + s.math) / 2.0;

        fseek(fp, (s.id - 1) * sizeof(struct student), SEEK_SET);

        fwrite(&s, sizeof(struct student), 1, fp);
    }

    printf("\n資料寫入完成\n\n");

    rewind(fp);

    printf("學生資料如下:\n");
    printf("學號\t姓名\t計概\t數學\t平均\n");

    while (fread(&s, sizeof(struct student), 1, fp) == 1) {

        if (s.id != 0) {
            printf("%d\t%s\t%d\t%d\t%.2f\n",
                s.id, s.name, s.computer, s.math, s.avg);
        }

    }

    fclose(fp);

    return 0;
}
