#include <stdio.h>
#include <string.h>
#define SIZE 10
void copy1(char* s1, const char* s2)
{
    for (; ((*s1 = *s2) != '\0'); ++s1, ++s2);
}
void swap_int(int* a, int* b) {
    int t = *a;
    *a = *b;
    *b = t;
}
void swap_float(float* a, float* b) {
    float t = *a;
    *a = *b;
    *b = t;
}
void swap_string(char a[], char b[]) {
    char temp[20];
    copy1(temp, a);
    copy1(a, b);
    copy1(b, temp);
}
void sort_array(char name[][20], int stu[][SIZE], float avg[], int n) {
    int i, j;
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - 1 - i; j++) {
            if (avg[j] < avg[j + 1]) {
                swap_string(name[j], name[j + 1]);
                swap_int(&stu[0][j], &stu[0][j + 1]);
                swap_int(&stu[1][j], &stu[1][j + 1]);
                swap_int(&stu[2][j], &stu[2][j + 1]);
                swap_float(&avg[j], &avg[j + 1]);
            }
        }
    }
}

int main() {
    const char* stu_name[SIZE] = {
        "Danny","Mary","Jimmy","Peter","Sue",
        "John","Hearts","Diamonds","Clubs","Spades"
    };

    char name[SIZE][20];
    for (int i = 0; i < SIZE; i++) {
        copy1(name[i], stu_name[i]);
    }

    int stu[3][SIZE] = {
        { 11, 15, 13, 18, 12, 17, 14, 19, 16, 20 }, 
        { 85, 92, 78, 85, 90, 85, 76, 95, 88, 85 }, 
        { 70, 80, 75, 90, 85, 70, 82, 92, 88, 70 }  
    };

    float average_scores[SIZE];

    for (int i = 0; i < SIZE; i++) {
        average_scores[i] = (stu[1][i] + stu[2][i]) / 2.0f;
    }
    printf("排序前：\n");
    printf("姓名\t學號\t計概成績\t數學成績\t平均成績\n");
    for (int i = 0; i < SIZE; i++) {
        printf("%-10s\t%2d\t   %3d\t\t   %3d\t\t  %.2f\n",
            name[i], stu[0][i], stu[1][i], stu[2][i], average_scores[i]);
    }
    sort_array(name, stu, average_scores, SIZE);
    printf("\n排序後（依平均成績由大到小）：\n");
    printf("姓名\t學號\t計概成績\t數學成績\t平均成績\n");
    for (int i = 0; i < SIZE; i++) {
        printf("%-10s\t%2d\t   %3d\t\t   %3d\t\t  %.2f\n",
            name[i], stu[0][i], stu[1][i], stu[2][i], average_scores[i]);
    }

    return 0;
}
