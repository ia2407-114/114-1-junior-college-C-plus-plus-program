#include <iostream>
#include <cmath> 

using namespace std;

int volume(int s) {
    cout << "正方體 (邊長 " << s << ") 的體積是: ";
    return s * s * s; 
}

double volume(int r, int h) {
    const double PI = 3.141592653589793;
    cout << "圓柱體 (半徑 " << r << ", 高 " << h << ") 的體積是: ";
    return PI * r * r * h;
}

int volume(int l, int w, int h) {
    cout << "長方體 (長 " << l << ", 寬 " << w << ", 高 " << h << ") 的體積是: ";
    return l * w * h;
}

int main() {
    cout << volume(7) << endl;

    cout << volume(5, 10) << endl;

    cout << volume(5, 6, 7) << endl;

    return 0;
}
