#include <stdio.h>
double approximate_pi(int n) {
    int i;
    double pi_approx = 0.0;
    double sign = 1.0;
    for (i = 0; i < n; i++) {
        double denominator = 2.0 * i + 1.0;
        pi_approx += sign * (4.0 / denominator);
        sign *= -1.0;
    }

    return pi_approx;
}

double circumference(double pi, double r) {
    return 2 * pi * r;
}
double circleArea(double pi, double r) {
    return pi * r * r;
}
double sphereVolume(double pi, double r) {
    return (4.0 / 3.0) * pi * r * r * r;
}
double sphereArea(double pi, double r) {
    return 4.0 * pi * r * r;
}
double cylinderVolume(double pi, double r, double h) {
    return pi * r * r * h;
}
double cylinderArea(double pi, double r, double h) {
    return 2 * pi * r * (r + h);
}
int main() {
    int n;
    double r, h;

    printf("請輸入計算項目總數 n（越大越準確）：");
    scanf_s("%d", &n);

    printf("請輸入半徑 r：");
    scanf_s("%lf", &r);

    printf("請輸入高 h（圓柱用）：");
    scanf_s("%lf", &h);
    double pi = approximate_pi(n);
    printf("\n近似的 π = %.10lf\n", pi);
    printf("圓周長 = %.4lf\n", circumference(pi, r));
    printf("圓面積 = %.4lf\n", circleArea(pi, r));
    printf("球體積 = %.4lf\n", sphereVolume(pi, r));
    printf("球表面積 = %.4lf\n", sphereArea(pi, r));
    printf("圓柱體積 = %.4lf\n", cylinderVolume(pi, r, h));
    printf("圓柱表面積 = %.4lf\n", cylinderArea(pi, r, h));
    return 0;
}
