#include <stdio.h>
double BMI(double height, double weight) {
    double bmi;
    bmi = weight / (height * height); // BMI公式
    return bmi; // 回傳BMI值
}
int main() {
    double height, weight, bmi;
    printf("請輸入身高(公尺)：");
    scanf_s("%lf", &height);
    printf("請輸入體重(公斤)：");
    scanf_s("%lf", &weight);
    bmi = BMI(height, weight);
    printf("您的BMI為：%.2lf\n", bmi);
    if (bmi < 18.5)
        printf("體重過輕\n");
    else if (bmi < 24)
        printf("體重正常\n");
    else if (bmi < 27)
        printf("體重過重\n");
    else
        printf("肥胖\n");
    return 0;
}
