#include <stdio.h>

int main() {
    int n;
    int i;
    double pi = 0.0;
    double sign = 1.0; 
    printf("請輸入項目數量 n：");
    scanf_s("%d", &n);
    for (i = 0; i < n; i++) {
        if (i < 0) {
            continue;  
        }
        double denominator = 2.0 * i + 1.0; 
        pi += sign * (4.0 / denominator); 
        sign = -sign;  
    }
    printf("pi 的近似值 = %.10f\n", pi);
    return 0;
}
