#include <stdio.h>

int main() {
    int a, b, c;
    int max, min;
    double avg;
    printf("請輸入三個整數：");
    scanf_s("%d %d %d", &a, &b, &c);
    int sum = a + b + c;
    avg = sum / 3.0;
    if (a >= b && a >= c)
        max = a;
    else if (b >= a && b >= c)
        max = b;
    else
        max = c;
    if (a <= b && a <= c)
        min = a;
    else if (b <= a && b <= c)
        min = b;
    else
        min = c;
    printf("總和 = %d\n", sum);
    printf("平均值 = %.2f\n", avg);
    printf("最大值 = %d\n", max);
    printf("最小值 = %d\n", min);

    return 0;
}
