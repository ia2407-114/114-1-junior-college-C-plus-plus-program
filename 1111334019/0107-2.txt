#include <stdio.h>
#include <string.h>

#define SIZE 10

struct student {
    char name[20];
    int id;
    int computer;
    int math;
    double avg;
};
void swap_student(struct student* a, struct student* b) {
    struct student temp = *a;
    *a = *b;
    *b = temp;
}
void sort_by_id(struct student stu[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1 - i; j++) {
            if (stu[j].id > stu[j + 1].id) {
                swap_student(&stu[j], &stu[j + 1]);
            }
        }
    }
}
int binary_search(struct student stu[], int n, int key) {
    int left = 0;
    int right = n - 1;

    while (left <= right) {
        int mid = (left + right) / 2;

        if (stu[mid].id == key)
            return mid;
        else if (stu[mid].id < key)
            left = mid + 1;
        else
            right = mid - 1;
    }

    return -1;
}

int main(void) {
    struct student stu[SIZE] = {
        {"Danny",    11, 85, 70, 0},
        {"Mary",     15, 92, 80, 0},
        {"Jimmy",    13, 78, 75, 0},
        {"Peter",    18, 85, 90, 0},
        {"Sue",      12, 90, 85, 0},
        {"John",     17, 85, 70, 0},
        {"Hearts",   14, 76, 82, 0},
        {"Diamonds", 19, 95, 92, 0},
        {"Clubs",    16, 88, 88, 0},
        {"Spades",   20, 85, 70, 0}
    };
    for (int i = 0; i < SIZE; i++) {
        stu[i].avg = (stu[i].computer + stu[i].math) / 2.0;
    }
    sort_by_id(stu, SIZE);

    int search_id;
    printf("請輸入要搜尋的學號：");
    scanf_s("%d", &search_id);

    int index = binary_search(stu, SIZE, search_id);

    if (index != -1) {
        printf("\n找到學生資料：\n");
        printf("姓名：%s\n", stu[index].name);
        printf("學號：%d\n", stu[index].id);
        printf("計概：%d\n", stu[index].computer);
        printf("數學：%d\n", stu[index].math);
        printf("平均：%.2f\n", stu[index].avg);
    }
    else {
        printf("\n查無此學號！\n");
    }

    return 0;
}
