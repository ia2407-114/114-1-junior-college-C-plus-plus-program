// Fig. 9.22: fig09_22.cpp  
// Friends can access private members of a class.
#include <iostream>
#include "Count.h"
using namespace std;

// printDate class definition
class printDate
{
public:
   void printX( Count &c )
   {
      cout << c.x << endl;
   } // end function printX
}; // end class printDate

// function setX can modify private data of Count
// because setX is declared as a friend of Count (line 9)
void setX( Count &c, int val )
{
   c.x = val; // allowed because setX is a friend of Count
} // end function setX

int main()
{
   Count counter; // create Count object
   printDate pd; // create printDate object

   cout << "counter.x after instantiation: ";
   pd.printX( counter );

   setX( counter, 8 ); // set x using a friend function
   cout << "counter.x after call to setX friend function: ";
   pd.printX( counter );
} // end main
