// Define class GradeBook that contains a courseName data member
// and member functions to set and get its value;
// Create and manipulate a GradeBook object with these functions.
#include <iostream>
#include <string> // program uses C++ standard string class
#include "pet.h"
using namespace std;

// function main begins program execution
int main()
{
   PET myPet( "Lucy", 3 ); // create a GradeBook object named myGradeBook
   PET myPet2( "Tom", 5 );
   // display initial value of courseName
   cout << "pet name is: " << myPet.getPetName()<< endl;
   cout << "pet age is: " << myPet.getPetAge()<< endl;

   cout << endl; // outputs a blank line
   myPet.displayMessage(); // display message with new course name
   cout << "pet name is: " << myPet2.getPetName()<< endl;
   cout << "pet age is: " << myPet2.getPetAge()<< endl;

   cout << endl; // outputs a blank line
   myPet2.displayMessage(); // display message with new course name
} // end main
