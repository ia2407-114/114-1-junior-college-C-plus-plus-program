#include<stdio.h>

int main(void)
{
	int chinese, english, cs, total;
	double avg;

	
	printf("請輸入你的國文成績:");
	scanf_s("%d", &chinese);

	printf("請輸入你的數學成績:");
	scanf_s("%d", &english);

	printf("請輸入你的計概成績:");
	scanf_s("%d", &cs);
	
	total = chinese + english + cs;
	avg = total / 3.0;

	printf("你的總分是:%d\n",total);
	printf("你的平均成績是:%.2f\n", avg);

	if (avg>=60)
	{
		printf("恭喜通過!\n");
	}
	else
	{
		printf("再加油!\n");
	}
	return 0;
}
