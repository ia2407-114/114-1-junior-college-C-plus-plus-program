#include <stdio.h>

int main()
{
    int a, b, c, d;
    int max, second, third, min;
    int mid1, mid2;

    printf("請輸入四個整數：");
    scanf_s("%d %d %d %d", &a, &b, &c, &d);

    max = a;
    if (b > max) max = b;
    if (c > max) max = c;
    if (d > max) max = d;

    min = a;
    if (b < min) min = b;
    if (c < min) min = c;
    if (d < min) min = d;

    mid1 = 0;
    mid2 = 0;

    if (a != max) {
        if (a != min) {
            mid1 = a;
        }
    }

    if (b != max) {
        if (b != min) {
            if (mid1 == 0) {
                mid1 = b;
            }
            else {
                mid2 = b;
            }
        }
    }

    if (c != max) {
        if (c != min) {
            if (mid1 == 0) {
                mid1 = c;
            }
            else if (mid2 == 0) {
                mid2 = c;
            }
        }
    }

    if (d != max) {
        if (d != min) {
            if (mid1 == 0) {
                mid1 = d;
            }
            else {
                mid2 = d;
            }
        }
    }


    if (mid1 > mid2) {
        second = mid1;
        third = mid2;
    }
    else {
        second = mid2;
        third = mid1;
    }

    printf("由大到小排序結果為：%d %d %d %d\n", max, second, third, min);

    return 0;
}
