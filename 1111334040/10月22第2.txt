#include <stdio.h>

int main()
{
    double p, r, deposit, aims;
    double balance;
    int years = 0;

    printf("輸入本金: ");
    scanf_s("%lf", &p);
    printf("輸入年利率: ");
    scanf_s("%lf", &r);
    printf("輸入每年定期定額存入金額: ");
    scanf_s("%lf", &deposit);
    printf("輸入目標金額: ");
    scanf_s("%lf", &aims);

    balance = p;

    while (balance < aims) {
        years++;
        balance = balance * (1.0 + r) + deposit;
        printf("第 %d 年底: %.2f\n", years, balance);
    }

    printf("達成目標 %.2f，共花 %d 年\n", aims, years);

    return 0;
}
