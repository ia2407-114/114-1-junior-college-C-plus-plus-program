#include<stdio.h>
double approximate_pi(int);
double c1(double, double);
double c2(double, double);
double c3(double, double);
double c4(double, double);
double c5(double, double);
double c6(double, double);
int main(void) {
    double r,h;
    int n;
    printf("請輸入pi取的近似值n:\n");
    scanf_s("%d", &n);
    printf("請輸入半徑r:\n");
    scanf_s("%lf", &r);
    printf("請輸入圓的高h:\n");
    scanf_s("%lf", &h);
    printf("圓周長%lf:\n",c1(approximate_pi(n),r));
    printf("圓面積長%lf:\n", c2(approximate_pi(n), r));
    printf("球體積%lf:\n", c3(approximate_pi(n), r));
    printf("球表面積%lf:\n", c4(approximate_pi(n), r));
    printf("圓柱體積%lf:\n", c5(approximate_pi(n), r,h));
    printf("圓柱表面積%lf:\n", c6(approximate_pi(n), r,h));
}
double approximate_pi(int n) {
    int i;
    double pi_approx = 0.0;
    double sign = 1.0;

    for (i = 0; i < n; i++) {
        double denominator = 2.0 * i + 1.0;
        pi_approx += sign * (4.0 / denominator);
        sign *= -1.0;
    }

    return pi_approx;
}
double c1(double pi, double r)
{
    return pi*r*2;
}
double c2(double pi, double r)
{
    return pi * r * r;
}
double c3(double pi, double r)
{
    return (4/3)*pi * r * r*r;
}
double c4(double pi, double r)
{
    return 4*pi * r ;
}
double c5(double pi, double r,double h)
{
    return pi * r * r* h;
}
double c6(double pi, double r,double h)
{
    return (2*pi*r*r)+(2*pi*r*h);
}
