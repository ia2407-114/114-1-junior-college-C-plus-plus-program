#include <stdio.h>

float a(float aims, float years, float r, float deposit, float c);

int main()
{
	float aims, years = 0, r, deposit, c;
	printf("請輸入本金\n");
	scanf_s("%f", &c);
	printf("請輸入利率\n");
	scanf_s("%f", &r);
	printf("請輸入目標金額\n");
	scanf_s("%f", &aims);
	printf("請輸入每一年底要再存多少錢進戶頭\n");
	scanf_s("%f", &deposit);
	printf("需要%.0f年", a(aims, years, r, deposit, c));
}
float a(float aims, float years, float r, float deposit, float c)
{
	while (c < aims)
	{
		years += 1;
		c = c + deposit;
		c = c * (1 + r);

	}
	return years;

}
