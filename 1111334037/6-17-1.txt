#include "Header.h"

void writeMovie()
{
    cinema movie;

    ofstream writebinaryfile("movie.bin",ios::out | ios::binary);

    if (writebinaryfile.fail())
    {
        cout << "movie.bin檔案無法開啟!\n";
        exit(1);
    }

    cout << "請輸入5筆電影資料\n";

    for (int i = 0; i < 5; i++)
    {
        cout << "\n第" << i + 1 << "筆資料\n";

        cout << "電影名稱:";
        cin >> movie.name;

        cout << "上映日期:";
        cin >> movie.date;

        cout << "上映廳處:";
        cin >> movie.place;

        cout << "票價:";
        cin >> movie.price;

        writebinaryfile.write((char*)&movie,sizeof(cinema));
    }

    writebinaryfile.close();
}

void showMovieList()
{
    cinema movie;

    ifstream readbinaryfile("movie.bin",ios::in | ios::binary);

    if (readbinaryfile.fail())
    {
        cout << "movie.bin檔案無法開啟!\n";
        exit(1);
    }

    cout << "\n電影資料清單\n";

    int i = 1;

    while (
        readbinaryfile.read((char*)&movie,sizeof(cinema)))
    {
        cout << i
            << ". "
            << movie.name
            << endl;

        i++;
    }

    readbinaryfile.close();
}

void showMovieDetail(int no)
{
    cinema movie;

    ifstream readbinaryfile("movie.bin",ios::in | ios::binary);

    if (readbinaryfile.fail())
    {
        cout << "movie.bin檔案無法開啟!\n";
        exit(1);
    }

    readbinaryfile.seekg(sizeof(cinema) * (no - 1),ios::beg);

    readbinaryfile.read( (char*)&movie,sizeof(cinema));

    if (!readbinaryfile.eof())
    {
        cout << "\n電影詳細資料\n";

        cout << "電影名稱:"
            << movie.name
            << endl;

        cout << "上映日期:"
            << movie.date
            << endl;

        cout << "上映廳處:"
            << movie.place
            << endl;

        cout << "票價:"
            << movie.price
            << endl;
    }
    else
    {
        cout << "查無資料\n";
    }

    readbinaryfile.close();
}實作
