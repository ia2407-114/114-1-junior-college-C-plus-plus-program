#include "Header.h"
#include <iomanip>

int main()
{
    Animal animal[3];

    ofstream writefile("animal.txt");

    if (writefile.fail())
    {
        cout << "檔案開啟失敗！" << endl;
        return 1;
    }

    for (int i = 0; i < 3; i++)
    {
        cout << "\n第 " << i + 1 << " 筆資料\n";
        animal[i].input();
    }


    cout << "\n=== 動物資料 ===\n";

    cout << left
        << setw(10) << "名稱"
        << setw(10) << "年齡"
        << setw(10) << "身高"
        << endl;

    for (int i = 0; i < 3; i++)
    {
        animal[i].output();
    }

    
    writefile << left
        << setw(10) << "名稱"
        << setw(10) << "年齡"
        << setw(10) << "身高"
        << endl;

    for (int i = 0; i < 3; i++)
    {
        animal[i].writeFile(writefile);
    }

    writefile.close();

    return 0;
}主程式

{
