double approximate_pi(int n) {
    int i;
    double pi_approx = 0.0;
    double sign = 1.0;

    for (i = 0; i < n; i++) {
        double denominator = 2.0 * i + 1.0;
        pi_approx += sign * (4.0 / denominator);
        sign *= -1.0;
    }

    return pi_approx;
}

/* 1. 利用傳入的整數 n 求出 pi 的近似值 (使用 continue) */
double approximate_pi(int n)
{
    int i;
    double pi_approx = 0.0;

    for (i = 0; i < n; i++)
    {
        double denominator = 2.0 * i + 1.0;

        if (denominator == 0)  // 不可能發生，但展示 continue 用法
            continue;

        if (i % 2 == 0)
            pi_approx += (4.0 / denominator);
        else
            pi_approx -= (4.0 / denominator);
    }

    return pi_approx;
}

/* 2. 圓周長 */
double circleCircumference(double pi, double r)
{
    return 2.0 * pi * r;
}

/* 3. 圓面積 */
double circleArea(double pi, double r)
{
    return pi * r * r;
}

/* 4. 球體積 */
double sphereVolume(double pi, double r)
{
    return (4.0 / 3.0) * pi * pow(r, 3);
}

/* 5. 球表面積 */
double sphereArea(double pi, double r)
{
    return 4.0 * pi * r * r;
}

/* 6. 圓柱體積 */
double cylinderVolume(double pi, double r, double h)
{
    return pi * r * r * h;
}

/* 7. 圓柱表面積 */
double cylinderArea(double pi, double r, double h)
{
    return 2.0 * pi * r * (r + h);
}

/* 建立與呼叫使用者自訂函數 */
#include <stdio.h>

/* 1. 函數的原型宣告 (prototype) */
int calcYears(double p, double r, double deposit, double aims);

/* 在 main() 函數呼叫 calcYears() 函數 */
int main()
{
    double p;        // 本金
    double r;        // 年利率
    double deposit;  // 每年定期定額存入金額
    double aims;     // 存款目標
    int years;       // 所需年數

    printf("請輸入本金：");
    scanf_s("%lf", &p);

    printf("輸入年利率：");
    scanf_s("%lf", &r);

    printf("輸入每年定期定額存入金額：");
    scanf_s("%lf", &deposit);

    printf("輸入存款目標金額：");
    scanf_s("%lf", &aims);

    /* 3. 函數的呼叫 */
    years = calcYears(p, r, deposit, aims);

    printf("\n達成目標共需 %d 年。\n", years);

    return 0;
}

/* 2. 函數的定義 */
int calcYears(double p, double r, double deposit, double aims)
{
    int years = 0;
    printf("\n年份\t本利合\n");

    while (p < aims)
    {
        p = p * (1 + r) + deposit;  // 複利 + 定期定額
        years++;
        printf("%d\t%.2lf\n", years, p);
    }

    return years;
}
