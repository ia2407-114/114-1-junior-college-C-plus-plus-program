#include <stdio.h>

/* 1. 函數的原型宣告 (Prototype) */
double calcBMI(double weight, double height); // 計算BMI的函數
void showResult(double bmi);                   // 顯示BMI分類結果的函數

/* 在 main() 函數中呼叫 calcBMI() 和 showResult() */
int main(void)
{
    double weight, height, bmi;

    printf("請輸入體重(公斤): ");
    scanf_s("%lf", &weight);

    printf("請輸入身高(公尺): ");
    scanf_s("%lf", &height);

    /* 3. 呼叫 calcBMI() 計算 BMI */
    bmi = calcBMI(weight, height);

    printf("你的BMI是: %.2f\n", bmi);

    /* 3. 呼叫 showResult() 顯示判斷結果 */
    showResult(bmi);

    return 0;
}

/* 2. 函數的定義區 */

/* 計算BMI的函數 */
double calcBMI(double weight, double height)
{
    return weight / (height * height);
}

/* 顯示BMI結果的函數 */
void showResult(double bmi)
{
    if (bmi < 18.5)
        printf("體重過輕\n");
    else if (bmi < 25.0)
        printf("正常體重\n");
    else if (bmi < 30.0)
        printf("過重\n");
    else if (bmi < 35.0)
        printf("輕度肥胖\n");
    else if (bmi < 40.0)
        printf("中度肥胖\n");
    else
        printf("重度肥胖(病態肥胖)\n");
}

第三題

承上一周的題目2，寫一個副程式，傳入本金int (p=10000) 年利率double (r=0.05)，每個年底定期定額存入金額int ，以及存款目標int(aims)，.傳回年數，在主程式印出要存幾年才能達成目標

   上週題目:從鍵盤輸入本金(p=10000) 年利率(r=0.05)，每個年底定期定額存入金額，以及存款目標(aims)印出每年底(複利)本利合 直到達成存款目標，讓使用者知道幾年才能達成目標
*****注意! 計算次方項時，不得使用庫存函數pow( , )!


/* 建立與呼叫使用者自訂函數 */
#include <stdio.h>

/* 1. 函數的原型宣告 (prototype) */
int calcYears(double p, double r, double deposit, double aims);

/* 在 main() 函數呼叫 calcYears() 函數 */
int main()
{
    double p;        // 本金
    double r;        // 年利率
    double deposit;  // 每年定期定額存入金額
    double aims;     // 存款目標
    int years;       // 所需年數

    printf("請輸入本金：");
    scanf_s("%lf", &p);

    printf("輸入年利率：");
    scanf_s("%lf", &r);

    printf("輸入每年定期定額存入金額：");
    scanf_s("%lf", &deposit);

    printf("輸入存款目標金額：");
    scanf_s("%lf", &aims);

    /* 3. 函數的呼叫 */
    years = calcYears(p, r, deposit, aims);

    printf("\n達成目標共需 %d 年。\n", years);

    return 0;
}

/* 2. 函數的定義 */
int calcYears(double p, double r, double deposit, double aims)
{
    int years = 0;
    printf("\n年份\t本利合\n");

    while (p < aims)
    {
        p = p * (1 + r) + deposit;  // 複利 + 定期定額
        years++;
        printf("%d\t%.2lf\n", years, p);
    }

    return years;
}
